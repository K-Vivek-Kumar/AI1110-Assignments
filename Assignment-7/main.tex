	\section{Papoulis-Exercise-5} \par\textbf{\underline{Question 5-7:}}  We place at random 200 points in the interval (0, 100). The distance from 0 to the first random point is a random variable $z$. Find $F_{z}(z)$.
	\begin{enumerate}
	\item exactly, and 
	\item using the Poisson approximation. 
	\end{enumerate}
	\par\textbf{\underline{Solution:} }Clearly, $z' \leq z$ iff the number n$(0,z)$ of the points in the interval $(0,z)$ is atleast one.\\
	Hence,
	\begin{align}
	F_{z}(z) &= \text{P}\cbrak{z' \leq z} \\
	         &= \text{P}\cbrak{\text{n'}(0,z)>0}\\
	F_{z}(z) &= 1-\text{P}\cbrak{\text{n'}(0,z)=0}
	\end{align}
	The probability p that a particular point is in the interval $(0,z)$ equals $\frac{z}{100}$. With $n=200,k=0$ and $p=\frac{z}{100}$, from the fundamental theorem \\
	\\
	\fbox{\begin{minipage}{20.5em}
\textbf{FUNDAMENTAL THEOREM:} 
\begin{align}
p_{n}(k)&=\text{P($A$ occurs $k$ times in any order)} \\
        &= \begin{pmatrix}
n\\
k
\end{pmatrix}p^{k}q^{n-k}
\end{align}
\end{minipage}}\\
\\
	it yields that $\text{P}\cbrak{\text{n'}(0,z)=0}=\brak{1-p}^{200}$. Hence,
	\begin{enumerate}
	\item \begin{align}
	F_{z}(z) &= 1-\brak{1-\frac{z}{100}}^{200}
	\end{align}
	\item From the following theorem statement,\\
	\\
	\fbox{\begin{minipage}{20.5em}
\textbf{POISSON THEOREM:} If 
\begin{align*}
n \to \infty; && p \to 0; && \text{such that } np \to \lambda
\end{align*}
then for $k=0,1,2..$,
\begin{align}
\frac{n!}{k!(n-k)!}p^{k}q^{n-k}\xrightarrow[\text{$n\to \infty$}]{} e^{-\lambda}\frac{\lambda^{k}}{k!}
\end{align}
\end{minipage}}\\
\\
	It follows that for $z \ll 100$,
	\begin{align}
	F_{z}(z) \simeq 1-e^{-2z}
\end{align}
	\end{enumerate}
